# C++
```
class Solution {
public:
    int maxProduct(vector<string>& words) {

        int n = words.size(), ans = 0;
        vector<int> bs(n);

        for(int i = 0;i < n;i++)
            for(auto c : words[i])
                bs[i] |= 1 << (c - 'a');

        for(int i = 0;i < n;i++)
            for(int j = i + 1;j < n;j++)
                if((bs[i] & bs[j]) == 0) 
                    ans = max(ans, int(words[i].length() * words[j].length()));
                    
        return ans;
    }
};
```
